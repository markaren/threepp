
function(add_test_executable name)
    add_executable(${name} ${name}.cpp)
    target_link_libraries(${name} PRIVATE threepp Catch2::Catch2)
    add_test(NAME ${name} COMMAND ${name})
endfunction()

add_test_executable(constants_test)
add_test_executable(Uniform_test)
add_test_executable(Vector3_test)
add_test_executable(Matrix4_test)
add_test_executable(Quaternion_test)
add_test_executable(Object3D_test)
add_test_executable(PerspectiveCamera_test)
add_test_executable(StringUtils_test)
add_test_executable(EventDispatcher_test)
target_include_directories(StringUtils_test PRIVATE "${CMAKE_SOURCE_DIR}/src")
